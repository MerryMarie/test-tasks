SELECT `likes`.`id_article`, COUNT(`likes`.`id`) as `likes_total`, `article_comments`.`comment_total` FROM `likes`, (SELECT `id_article` FROM `authors_articles` WHERE `id_author`=2) as `author_articles` JOIN (SELECT `id_article`, count(*) as `comment_total` FROM `articles_comments` GROUP BY `id_article`) as `article_comments` on `article_comments`.`id_article` = `author_articles`.`id_article`  WHERE `likes`.`id_article`=`author_articles`.`id_article` GROUP BY `likes`.`id_article`;
